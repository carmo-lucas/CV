\newcommand{\entry}[6]{
% 1 - Data final
% 2 - Data inicial
% 3 - Cargo
% 4 - Empresa / Instituição
% 5 - Descrição
% 6 - Local

	\noindent \large{#3} \hfill \small{\textcolor{Accent}{#6}} \\
	\textcolor{Accent}{\textbf{\textsc{#4}}} \hfill \textsc{#2} \textcolor{Accent}{-\textrightarrow} \textsc{#1} \\
	\textcolor{Accent}{\vline} \hspace*{0.1cm}
	\begin{minipage}[t]{.75\textwidth}
		\footnotesize{#5}
	\end{minipage}
	\vspace*{.5cm}
}

\newcommand{\skill}[4]{
	% 1. Skill
	% 2. Proficiency
	% 3. Time spent learning
	% 4. Description
	\noindent
	\begin{minipage}[t]{.45\textwidth}
		\noindent #1  
		\footnotesize\scshape\color{gray}		
		\ifthenelse{\equal{#2}{1}}{
			\highlight{\CIRCLE} \CIRCLE \CIRCLE \CIRCLE \CIRCLE 
		}{}
		\ifthenelse{\equal{#2}{2}}{
			\highlight{\CIRCLE \CIRCLE} \CIRCLE \CIRCLE \CIRCLE 
		}{}
		\ifthenelse{\equal{#2}{3}}{
			\highlight{\CIRCLE \CIRCLE \CIRCLE} \CIRCLE \CIRCLE 
		}{}
		\ifthenelse{\equal{#2}{4}}{
			\highlight{\CIRCLE \CIRCLE \CIRCLE \CIRCLE} \CIRCLE 
		}{}
		\ifthenelse{\equal{#2}{5}}{
			\highlight{\CIRCLE \CIRCLE \CIRCLE \CIRCLE \CIRCLE} 
		}{} \hfill #3 \\
		#4
	\end{minipage}
					
	\vspace*{0.5cm}
}
